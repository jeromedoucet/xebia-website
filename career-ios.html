#import <UIKit/UIKit.h>
#import "XBViewController.h"
 
@interface XBTableView : UITableView
 
@property(nonatomic, copy)dispatch_block_t onReloadData;
 
@end
 
@implementation XBTableView
 
- (void)reloadData {
    [super reloadData];
    self.onReloadData();
}
 
@end
 
@interface XBViewController () '<UITableViewDataSource>'
 
@end
 
@implementation XBViewController
 
- (void)loadView {
    XBTableView *view = [XBTableView new];
    view.dataSource = self;
 
    view.onReloadData = ^{
    self.title = @"Data up-to-date";
    };
 
    [view registerClass:[XBTableViewCell class] forCellReuseIdentifier:@"XBTableViewCell"];
 
    self.view = view;
    self.view.backgroundColor = [UIColor whiteColor];
    [view reloadData];
}
 
- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section {
    return 5;
}
 
- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath
{
    XBTableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:@"XBTableViewCell"];
    cell.textLabel.text = [NSString stringWithFormat:@"%ld", (long)indexPath.row];
    return cell;
}
 
@end

- (Vraiment) Pratiquer l’agilité au quotidien (Sprint, Code Review, Pair Programming, etc.);

- Rendre à la communauté ce qu’elle nous apporte en partageant son expérience à travers l’organisation de tech-events, la rédaction de billets de blog ou d’articles dans notre revue de presse hebdomadaire ou encore la participation à des conférences telles que Dotswift et Devoxx;
