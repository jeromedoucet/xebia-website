private class StringAdapter extends BaseAdapter {

       private final Context context;
       private final List"<"String">" strings = new ArrayList<>();

       private StringAdapter(Context context, List"<"String">" strings) {
           this.context = context;
           this.strings.addAll(strings);
       }

       @Override
       public int getCount() { return strings.size(); }

       @Override
       public String getItem(int position) { return strings.get(position); }

       @Override
       public long getItemId(int position) { return getItem(position).hashCode(); }

               @Override
